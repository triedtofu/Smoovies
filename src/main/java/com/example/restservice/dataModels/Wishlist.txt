package com.example.restservice.dataModels;
/**
 * Table called Wishlist where a user-id maps to a movie-id. A user's wishlist will be taken by 
 * getting all the pairs from a specific userid.
 */

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.ManyToOne;
import javax.persistence.MapsId;
import javax.persistence.Table;

import com.fasterxml.jackson.annotation.JsonProperty;

@Entity
@Table(name = "Wishlists")
public class Wishlist {
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @ManyToOne
    @MapsId
    @JoinColumn(name = "user_id")
    private User user;

    @ManyToOne
    @JoinColumn(name = "movie_id")
    private Movie movie;

    /**
     * Used to toggle for private wishlist.
     */
    @Column(name = "private")
    private Boolean isPrivate;

    public Wishlist() {
        super();
    }
    public Wishlist(@JsonProperty("user_id") Long user_id,
                    @JsonProperty("movie_id") Long movie_id,
                    @JsonProperty("isPrivate") Boolean isPrivate)
    {
        super();
    }

    public User getUser() {
        return this.user;
    }

    public Movie getMovie() {
        return this.movie;
    }

    public Boolean getIsPrivate() {
        return this.isPrivate;
    }
}
